#version 330 core

in vec4 color;
in vec2 TexCoord;

uniform sampler2D texture1;
uniform bool draw_texture;

out vec4 FragColor;

void main()
{
	if (draw_texture) {
		FragColor = texture(texture1, TexCoord) * vec4(color.r, color.g, color.b, 1.0f);
	}
	else {
		FragColor = vec4(color.r, color.g, color.b, 1.0f);
	}
}
